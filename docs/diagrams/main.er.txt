utente {
  id: int, not null, unique, PRIMARY
  
  first_name: string
  last_name: string
  age: int
  email: string
  password: string
  role: string  // 'ADMIN' o 'CUSTOMER'
  created_at: date
}

automobile {
  id: int, not null, unique, PRIMARY
  
  brand: string
  model: string
  license_plate: string
  year: int
  availability: boolean
  created_at: date
}

prenotazione {
  id: int, not null, unique, PRIMARY
  id_utente: int, not null, unique, FOREIGN
  id_automobile: int, not null, unique, FOREIGN

  startedAt: date
  endedAt: date
  status: string
  created_at: date
}


utente (0,n)<-->(1,n) prenotazione
prenotazione (n,n)<-->(0,1) automobile
